Remove PDF background (Image object) - For PPT#

The background of the PDF-PPT affects readability.

Requirement:

pip install pymupdf

import fitz  # PyMuPDF

def remove_images_from_pdf(input_pdf_path, output_pdf_path):
   
    pdf_document = fitz.open(input_pdf_path)

    for page_num in range(len(pdf_document)):
        page = pdf_document.load_page(page_num)

        image_list = page.get_images(full=True)

        if image_list:
            for img in image_list:
                xref = img[0]
                page.delete_image(xref)

    pdf_document.save(output_pdf_path)
    pdf_document.close()
    print(f" {output_pdf_path} already")

input_pdf = "input.pdf" 
output_pdf = "output_no_images.pdf" 
remove_images_from_pdf(input_pdf, output_pdf)

Convert to Black and White PDF - For Print#

The actual effect of printing color documents in black and white is very poor, and it requires black enhancement of PDF fonts. This method involves image grayscale conversion, and adjusting the DPI in the code can change the page clarity. The higher the DPI, the larger the file (this doesn’t matter for printing purposes, but other uses may require different methods).

(Example) Before PDF processing:

Requirement:

pip install pymupdf
pip install pdf2image pillow
brew install poppler

Set to Black and White (Image Method)#

from pdf2image import convert_from_path
from PIL import Image

def convert_pdf_to_grayscale(input_pdf, output_pdf, dpi=900): 
   
    images = convert_from_path(input_pdf, dpi=dpi)
    
    gray_images = []
    
    for img in images:
        gray_img = img.convert('L')
        gray_images.append(gray_img)

    gray_images[0].save(output_pdf, save_all=True, append_images=gray_images[1:])

input_pdf_path = "input.pdf"  
output_pdf_path = "output.pdf" 

convert_pdf_to_grayscale(input_pdf_path, output_pdf_path, dpi=900)  
print(f"Saved: {output_pdf_path}")

Gray text turns to bold#

from pdf2image import convert_from_path
from PIL import Image, ImageEnhance

def enhance_contrast_and_convert_to_bw(input_pdf, output_pdf, dpi=900, contrast_factor=2.0):

    images = convert_from_path(input_pdf, dpi=dpi)

    enhanced_images = []
    
    for img in images:

        gray_img = img.convert('L')
    
        enhancer = ImageEnhance.Contrast(gray_img)
        enhanced_img = enhancer.enhance(contrast_factor)

        enhanced_images.append(enhanced_img)
  
    enhanced_images[0].save(output_pdf, save_all=True, append_images=enhanced_images[1:])

input_pdf_path = "input.pdf"  
output_pdf_path = "output.pdf"  

enhance_contrast_and_convert_to_bw(input_pdf_path, output_pdf_path, dpi=900, contrast_factor=2.0)
print(f"Saved: {output_pdf_path}")

Invert black and white#

from pdf2image import convert_from_path
from PIL import Image
import numpy as np

def invert_colors_rgb(img):
  
    arr = np.array(img)

    inverted_arr = 255 - arr

    return Image.fromarray(inverted_arr, 'RGB')

def invert_pdf_colors(input_pdf, output_pdf, dpi=300):
    images = convert_from_path(input_pdf, dpi=dpi)

    inverted_images = []
    
    for img in images:
        inverted_img = invert_colors_rgb(img)
        
        inverted_images.append(inverted_img)
    
    inverted_images[0].save(output_pdf, save_all=True, append_images=inverted_images[1:])

input_pdf_path = "input.pdf" 
output_pdf_path = "output.pdf" 

invert_pdf_colors(input_pdf_path, output_pdf_path, dpi=300)
print(f"Saved：{output_pdf_path}")

Remove background gray dots#

from pdf2image import convert_from_path
from PIL import Image
import numpy as np

def clean_gray_dots(img, tolerance=5, min_brightness=50, max_brightness=200):
    arr = np.array(img)
    R = arr[:, :, 0].astype(int)
    G = arr[:, :, 1].astype(int)
    B = arr[:, :, 2].astype(int)

    diff_rg = np.abs(R - G)
    diff_rb = np.abs(R - B)
    diff_gb = np.abs(G - B)

    gray_mask = (diff_rg <= tolerance) & (diff_rb <= tolerance) & (diff_gb <= tolerance)

    brightness = (R + G + B) / 3

    brightness_mask = (brightness >= min_brightness) & (brightness <= max_brightness)

    gray_pixels = gray_mask & brightness_mask

    arr[gray_pixels] = [255, 255, 255]

    return Image.fromarray(arr.astype('uint8'), 'RGB')

def process_and_save_pdf_pages(input_pdf, output_pdf):
    images = convert_from_path(input_pdf, dpi=100)
    processed_images = []
    
    for img in images:
        cleaned_img = clean_gray_dots(img)
        processed_images.append(cleaned_img)
    
    processed_images[0].save(output_pdf, "PDF", save_all=True, append_images=processed_images[1:])

input_pdf_path = "input.pdf"
output_pdf_path = "output.pdf"

process_and_save_pdf_pages(input_pdf_path, output_pdf_path)
print(f"Saved：{output_pdf_path}")

PDF effective content clipping - For Print#

from pdf2image import convert_from_path
from PIL import Image
import numpy as np

def crop_pdf(input_pdf_path, output_pdf_path, dpi=200, margin=3):
    images = convert_from_path(input_pdf_path, dpi=dpi)
    cropped_images = []

    for image in images:
        img_array = np.array(image)

        if img_array.shape[2] == 4:
            img_array = img_array[:, :, :3]

        gray = np.mean(img_array, axis=2)

        coords = np.column_stack(np.where(gray < 250))

        if coords.size == 0:
            continue

        top_left = coords.min(axis=0) - margin
        bottom_right = coords.max(axis=0) + margin

        top_left = np.maximum(0, top_left)
        bottom_right = np.minimum([img_array.shape[0], img_array.shape[1]], bottom_right)

        cropped_image = image.crop((top_left[1], top_left[0], bottom_right[1], bottom_right[0]))

        cropped_images.append(cropped_image)

    if cropped_images:
        cropped_images[0].save(output_pdf_path, save_all=True, append_images=cropped_images[1:])
    else:
        print("None")


input_pdf_path = "input.pdf"
output_pdf_path = "output.pdf"
crop_pdf(input_pdf, output_pdf)
